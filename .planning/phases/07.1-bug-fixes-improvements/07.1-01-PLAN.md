         ```
         function isQuotaOrRateLimitError(err: unknown): boolean {
           const msg = err instanceof Error ? err.message : String(err);
           return (
             msg.includes("quota") ||
             msg.includes("RESOURCE_EXHAUSTED") ||
             msg.includes("429") ||
             msg.includes("rate limit") ||
             msg.includes("Too Many Requests")
           );
         }
         ```
         Define this function above the POST handler. In the catch block, check `if (isQuotaOrRateLimitError(err))` and return:
         ```
         return Response.json(
           { error: "AI recommendations are temporarily unavailable. Please try again later." },
           { status: 503 }
         );
         ```
       - Also check the cause chain recursively: if `err` has a `cause` property, call `isQuotaOrRateLimitError` on it too.
