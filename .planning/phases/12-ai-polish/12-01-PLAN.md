  ```typescript
  // Pick a redirect message deterministically (avoid Math.random in server context)
  const message = REDIRECT_MESSAGES[Date.now() % REDIRECT_MESSAGES.length];

  const stream = createUIMessageStream({
    execute: ({ writer }) => {
      writer.write({ type: "text", text: message });
    },
  });

  return new Response(stream.value, {
    status: 200,
    headers: { "Content-Type": "text/event-stream" },
  });
  ```
