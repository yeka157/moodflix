**Step 2 — Generate the blurred movie poster collage (`public/og-base.png`):**

The user's locked decision requires a real movie poster collage as the OG background. Since Satori cannot apply CSS blur at runtime (open bug #573), the blurred image must be pre-generated offline and loaded from disk.

Create `scripts/generate-og-base.js`:

```javascript
const path = require("path");
const https = require("https");

// Popular movie backdrops from TMDB (publicly accessible CDN, no API key needed)
const BACKDROP_URLS = [
  "https://image.tmdb.org/t/p/w780/rAiYTfKGqDCRIIqo664sY9XZIvQ.jpg", // Interstellar
  "https://image.tmdb.org/t/p/w780/nMKdUUepR0i5zn0y1T4CejMOfAz.jpg", // The Dark Knight
  "https://image.tmdb.org/t/p/w780/jYEW5xZkZk2WTrdbMackzotvkgR.jpg", // Dune (2021)
  "https://image.tmdb.org/t/p/w780/ilRyazdMJwN05exqhwK4tMKBYZs.jpg", // Blade Runner 2049
];

function download(url) {
  return new Promise((resolve, reject) => {
    https.get(url, (res) => {
      if (res.statusCode === 301 || res.statusCode === 302) {
        return resolve(download(res.headers.location));
      }
      const chunks = [];
      res.on("data", (c) => chunks.push(c));
      res.on("end", () => resolve(Buffer.concat(chunks)));
      res.on("error", reject);
    }).on("error", reject);
  });
}

async function main() {
  // sharp is available in the Next.js install (no separate install needed)
  const sharp = require("sharp");
  const outFile = path.join(__dirname, "..", "public", "og-base.png");

  console.log("Downloading backdrops...");
  const buffers = await Promise.all(BACKDROP_URLS.map(download));

  console.log("Processing with sharp...");
  // Each backdrop: resize to 600x315, apply heavy blur, darken
  const CELL_W = 600, CELL_H = 315;
  const processed = await Promise.all(
    buffers.map((buf) =>
      sharp(buf)
        .resize(CELL_W, CELL_H, { fit: "cover", position: "centre" })
        .blur(28)
        .modulate({ brightness: 0.45 })
        .toBuffer()
    )
  );

  // Composite 4 cells into 1200x630 (2x2 grid)
  await sharp({
    create: { width: 1200, height: 630, channels: 4, background: { r: 10, g: 10, b: 10, alpha: 1 } },
  })
    .composite([
      { input: processed[0], top: 0, left: 0 },
      { input: processed[1], top: 0, left: 600 },
      { input: processed[2], top: 315, left: 0 },
      { input: processed[3], top: 315, left: 600 },
    ])
    .png()
    .toFile(outFile);

  console.log(`✓ Generated ${outFile}`);
}

main().catch((e) => { console.error(e); process.exit(1); });
```

**Step 3 — Create `app/opengraph-image.tsx` (global OG):**

CRITICAL Satori rules (violations produce silent failures or broken images):
- NO `display: grid` — use `display: flex` only
- NO `filter: blur()` — silently dropped (GitHub issue #573) — we use pre-blurred `og-base.png` instead
- NO React component functions inside ImageResponse JSX — inline all SVG/HTML as flat elements
- NO `oklch()` colors — use `#FB2C36` hex
- All elements need explicit `display: flex` if they contain children
- `position: absolute` works but requires parent to have `position: relative` and explicit `display: flex`

The background uses `public/og-base.png` — the pre-blurred real movie poster collage generated by the script above. This honors the user's locked decision: "Movie poster collage style: real movie posters (heavily blurred/darkened)".

```tsx
import { ImageResponse } from "next/og";
import { readFile } from "node:fs/promises";
import { join } from "node:path";

export const alt = "Moodflix - AI-Powered Movie Discovery";
export const size = { width: 1200, height: 630 };
export const contentType = "image/png";

export default async function Image() {
  // Load pre-blurred movie poster collage (generated by scripts/generate-og-base.js)
  const bgData = await readFile(join(process.cwd(), "public/og-base.png"));
  const bgSrc = `data:image/png;base64,${bgData.toString("base64")}`;

  // Load Bebas Neue font for Satori (cannot use next/font/google CSS variables here)
  const fontData = await readFile(
    join(process.cwd(), "assets/BebasNeue-Regular.woff2")
  );

  return new ImageResponse(
    (
      <div
        style={{
          width: "100%",
          height: "100%",
          display: "flex",
          alignItems: "center",
          justifyContent: "center",
          backgroundColor: "#0a0a0a",
          position: "relative",
        }}
      >
        {/* Pre-blurred movie poster collage background */}
        <img
          src={bgSrc}
          style={{
            position: "absolute",
            top: 0,
            left: 0,
            width: "100%",
            height: "100%",
            objectFit: "cover",
            opacity: 0.5,
          }}
        />

        {/* Dark vignette gradient overlay */}
        <div
          style={{
            position: "absolute",
            top: 0,
            left: 0,
            right: 0,
            bottom: 0,
            background:
              "linear-gradient(135deg, rgba(10,10,10,0.90) 0%, rgba(10,10,10,0.65) 50%, rgba(10,10,10,0.90) 100%)",
            display: "flex",
          }}
        />

        {/* Centered wordmark — LOGO ONLY, no subtitle (locked user decision) */}
        <div
          style={{
            position: "relative",
            display: "flex",
            flexDirection: "column",
            alignItems: "center",
            justifyContent: "center",
          }}
        >
          {/* MOODFLIX — crimson M + white OODFLIX */}
          <div
            style={{
              display: "flex",
              fontSize: 130,
              fontFamily: "BebasNeue",
              letterSpacing: "-3px",
              lineHeight: 1,
            }}
          >
            <span style={{ color: "#FB2C36" }}>M</span>
            <span style={{ color: "#ffffff" }}>OODFLIX</span>
          </div>

          {/* Thin crimson rule below wordmark */}
          <div
            style={{
              width: 80,
              height: 3,
              backgroundColor: "#FB2C36",
              marginTop: 16,
              display: "flex",
            }}
          />
        </div>
      </div>
    ),
    {
      ...size,
      fonts: [
        {
          name: "BebasNeue",
          data: fontData,
          style: "normal",
          weight: 400,
        },
      ],
    }
  );
}
```

Note: NO subtitle text in any OG image — only the wordmark with decorative rule. This is a locked user decision.
