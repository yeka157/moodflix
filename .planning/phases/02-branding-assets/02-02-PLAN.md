**Step 2 — Create `scripts/generate-icons.mjs`:**

The script renders the MoodflixIcon M path geometry as SVG strings at each size, then passes each through sharp for PNG output.

The icon has two variants for this pipeline:
- **Standard** (16, 32, 192, 512): M on transparent background. `cutoutColor` must visually work on any background — use `#0a0a0a` (the dark app bg). This makes the perforations invisible on dark but acceptable since most browser tabs are dark or the favicon sits in a colored context.
- **Solid bg** (apple-touch-icon 180px, maskable 512px): M on solid `#0a0a0a` background to ensure it looks good on iOS and Android home screens.
- **Maskable** (512px): M centered in 60% of canvas (so ~307px of 512px), with solid `#0a0a0a` background. The remaining 20% on each side is safe-zone padding for Android adaptive icon masks.

```js
// scripts/generate-icons.mjs
import sharp from "sharp";
import { writeFile } from "node:fs/promises";
import { join } from "node:path";

const crimson = "#FB2C36";
const dark = "#0a0a0a";

// SVG template for the M icon at a given canvas size
// For transparent variants: background rect omitted, cutout matches dark bg
// For solid bg variants: background rect added
function buildIconSvg({ canvas, iconSize, offsetX, offsetY, withBg, bgColor = dark }) {
  // Scale factor: the M geometry is defined in 0–100 viewBox
  // iconSize is the rendered pixel size of the M within the canvas
  const scale = iconSize / 100;

  // M path points scaled from the 100x100 viewBox
  // Original: M8 88 L8 12 L50 58 L92 12 L92 88 L80 88 L80 32 L50 70 L20 32 L20 88 Z
  // We use a nested SVG with its own viewBox to avoid manual coordinate math
  const bgRect = withBg ? `<rect width="${canvas}" height="${canvas}" fill="${bgColor}"/>` : "";

  return `<svg width="${canvas}" height="${canvas}" xmlns="http://www.w3.org/2000/svg">
  ${bgRect}
  <svg x="${offsetX}" y="${offsetY}" width="${iconSize}" height="${iconSize}" viewBox="0 0 100 100" fill="none">
    <path d="M8 88 L8 12 L50 58 L92 12 L92 88 L80 88 L80 32 L50 70 L20 32 L20 88 Z" fill="${crimson}"/>
    <rect x="10" y="18" width="8" height="10" rx="1.5" fill="${withBg ? bgColor : dark}"/>
    <rect x="10" y="34" width="8" height="10" rx="1.5" fill="${withBg ? bgColor : dark}"/>
    <rect x="10" y="50" width="8" height="10" rx="1.5" fill="${withBg ? bgColor : dark}"/>
    <rect x="10" y="66" width="8" height="10" rx="1.5" fill="${withBg ? bgColor : dark}"/>
  </svg>
</svg>`;
}

const icons = [
  // Standard favicons — transparent bg, M fills most of canvas
  {
    filename: "favicon-16x16.png",
    canvas: 16,
    iconSize: 16,
    offsetX: 0,
    offsetY: 0,
    withBg: false,
  },
  {
    filename: "favicon-32x32.png",
    canvas: 32,
    iconSize: 32,
    offsetX: 0,
    offsetY: 0,
    withBg: false,
  },
  // Apple touch icon — solid dark bg, M fills canvas with small inset
  {
    filename: "apple-touch-icon.png",
    canvas: 180,
    iconSize: 140,
    offsetX: 20,
    offsetY: 20,
    withBg: true,
  },
  // PWA icons — solid dark bg
  {
    filename: "icon-192.png",
    canvas: 192,
    iconSize: 150,
    offsetX: 21,
    offsetY: 21,
    withBg: true,
  },
  {
    filename: "icon-512.png",
    canvas: 512,
    iconSize: 400,
    offsetX: 56,
    offsetY: 56,
    withBg: true,
  },
  // Maskable — M must fit within center 60% of canvas (307px of 512)
  // Safe zone: 20% padding on each side = 102px offset, icon = 308px
  {
    filename: "icon-maskable.png",
    canvas: 512,
    iconSize: 308,
    offsetX: 102,
    offsetY: 102,
    withBg: true,
  },
];

const outDir = join(process.cwd(), "public");

for (const icon of icons) {
  const svg = buildIconSvg(icon);
  const buffer = Buffer.from(svg);
  const png = await sharp(buffer)
    .resize(icon.canvas, icon.canvas)
    .png()
    .toBuffer();
  await writeFile(join(outDir, icon.filename), png);
  console.log(`Generated: public/${icon.filename} (${icon.canvas}x${icon.canvas})`);
}

console.log("Done. All icons generated.");
```
