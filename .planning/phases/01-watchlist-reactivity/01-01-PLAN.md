export async function getWatchlistTmdbIds(): Promise<WatchlistTmdbEntry[]> {
  const userId = await getAuthUserId();
  if (!userId) return [];

  const rows = await db
    .select({ tmdbId: watchlist.tmdbId, status: watchlist.status })
    .from(watchlist)
    .where(eq(watchlist.userId, userId));

  return rows.map((r) => ({
    tmdbId: r.tmdbId,
    status: (r.status === "watching" ? "want_to_watch" : r.status) ?? "want_to_watch" as WatchlistStatus,
  }));
}

**4. Add `onMutate` to `useUpdateWatchlistStatus`** (lines 139-148):
Follow the `useRemoveFromWatchlist` pattern:
```typescript
onMutate: async (params) => {
  await queryClient.cancelQueries({ queryKey: watchlistKeys.all });

  // Update tmdbIds cache
  const previousTmdbIds = queryClient.getQueryData<WatchlistTmdbEntry[]>(watchlistKeys.tmdbIds());
  queryClient.setQueryData<WatchlistTmdbEntry[]>(watchlistKeys.tmdbIds(), (old) =>
    old?.map((entry) =>
      // Find by id through check cache is complex; simpler to invalidate
      entry
    ) ?? []
  );

  // Update check cache for the specific item
  const checkKeys = queryClient.getQueriesData<WatchlistItem | null>({
    queryKey: watchlistKeys.all,
    predicate: (query) => query.queryKey[1] === "check",
  });
  const previousChecks: Record<string, WatchlistItem | null | undefined> = {};
  for (const [key, data] of checkKeys) {
    if (data && data.id === params.id) {
      const keyStr = JSON.stringify(key);
      previousChecks[keyStr] = data;
      queryClient.setQueryData<WatchlistItem>(key, {
        ...data,
        status: params.status,
        watchedAt: params.status === "watched" ? new Date().toISOString() : null,
      });
      // Also update tmdbIds for this item
      queryClient.setQueryData<WatchlistTmdbEntry[]>(watchlistKeys.tmdbIds(), (old) =>
        old?.map((entry) =>
          entry.tmdbId === data.tmdbId ? { ...entry, status: params.status } : entry
        ) ?? []
      );
    }
  }

  // Update list caches
  const listKeys = [watchlistKeys.list(), watchlistKeys.list("want_to_watch"), watchlistKeys.list("watched")];
  const previousLists: Record<string, WatchlistItem[] | undefined> = {};
  for (const key of listKeys) {
    const keyStr = JSON.stringify(key);
    previousLists[keyStr] = queryClient.getQueryData<WatchlistItem[]>(key);
  }
  // Update "all" list
  queryClient.setQueryData<WatchlistItem[]>(watchlistKeys.list(), (old) =>
    old?.map((item) => item.id === params.id ? { ...item, status: params.status, watchedAt: params.status === "watched" ? new Date().toISOString() : null } : item) ?? []
  );
  // Move between status-filtered lists
  queryClient.setQueryData<WatchlistItem[]>(watchlistKeys.list("want_to_watch"), (old) => {
    if (params.status === "want_to_watch") {
      const item = previousLists[JSON.stringify(watchlistKeys.list())]?.find((i) => i.id === params.id);
      return item ? [...(old ?? []), { ...item, status: "want_to_watch", watchedAt: null }] : old ?? [];
    }
    return old?.filter((item) => item.id !== params.id) ?? [];
  });
  queryClient.setQueryData<WatchlistItem[]>(watchlistKeys.list("watched"), (old) => {
    if (params.status === "watched") {
      const item = previousLists[JSON.stringify(watchlistKeys.list())]?.find((i) => i.id === params.id);
      return item ? [...(old ?? []), { ...item, status: "watched", watchedAt: new Date().toISOString() }] : old ?? [];
    }
    return old?.filter((item) => item.id !== params.id) ?? [];
  });

  return { previousTmdbIds, previousChecks, previousLists };
},
onError: (_err, _params, context) => {
  // Restore all caches from context (same pattern as useRemoveFromWatchlist)
},
```
